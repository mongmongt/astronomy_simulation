    function computeBlackbodyColorMatrix(temp) {

      let calcT = temp / 100;

      let r, g, b;


      // Red 화소 축 환산

      if (calcT <= 66) {

        r = 255;

      } else {

        r = Math.min(255, Math.max(0, r));

      }


      // Green 화소 축 환산

      if (calcT <= 66) {

        g = calcT;

        g = 99.4708025861 * Math.log(g) - 161.1195681661;

        g = Math.min(255, Math.max(0, g));

      } else {

        g = calcT - 60;

        g = 288.1221695283 * Math.pow(g, -0.0755148492);

        g = Math.min(255, Math.max(0, g));

      }


      // Blue 화소 축 환산

      if (calcT >= 66) {

        b = 255;

      } else {

        if (calcT <= 19) {

          b = 0;

        } else {

          b = calcT - 10;

          b = 138.5177312231 * Math.log(b) - 305.0447927307;

          b = Math.min(255, Math.max(0, b));

        }

      }

        g: Math.round(g),

        b: Math.round(b),

    function evaluateHarvardType(temp) {

        subDigit = Math.floor(9 - ((temp - 30000) / 1111));

        subDigit = Math.floor(9 - ((temp - 10000) / 2222));

      } else if (temp >= 7500) {

        alphaClass = "A";

        subDigit = Math.floor(9 - ((temp - 7500) / 277));

      } else if (temp >= 6000) {

        alphaClass = "F";

        subDigit = Math.floor(9 - ((temp - 6000) / 166));

      } else if (temp >= 5200) {

        alphaClass = "G";

        subDigit = Math.floor(9 - ((temp - 5200) / 88));

      } else if (temp >= 3700) {

        alphaClass = "K";

      } else {

        alphaClass = "M";

        subDigit = Math.floor(9 - ((temp - 2000) / 188));

      }


      subDigit = Math.min(9, Math.max(0, subDigit));

      return `${alphaClass}${subDigit}`;

    }

    function renderStellarPhotosphere() {

      const currentR = parseFloat(radiusSlider.value);

      const currentT = parseFloat(tempSlider.value);


      ctx.clearRect(0, 0, canvas.width, canvas.height);


      const xCenter = canvas.width / 2;

      const yCenter = canvas.height / 2;


      // 1. 위상 공간 크기 가드레일 제어 연산

      let targetPixelRadius = 25;

      if (activeScalingMode === 'sqrt') {

        // 비선형 제곱근 차원 압축 축척 보정

        targetPixelRadius = 15 + (Math.sqrt(currentR / 50.0) * 125);

      } else {

        // 절대 기하 1:1 정비례 축척 매핑

        targetPixelRadius = Math.max(1, currentR * 3);

      }


      // 복사 온도 컬러 데이터 도출

      const computedRGB = computeBlackbodyColorMatrix(currentT);

      const rgbString = `${computedRGB.r}, ${computedRGB.g}, ${computedRGB.b}`;

      canvas.style.setProperty('--star-color', computedRGB.hex);


      // 2. 이중 렌더링 이펙트 - 파동 코로나 브리딩 애니메이션 필터

      wavePulseAngle += 0.03;

      const wavePulseOffset = Math.sin(wavePulseAngle) * 4.5;

      const dynamicCoronaRadius = targetPixelRadius + 52 + wavePulseOffset;


      // 외곽 소프트 코로나 플럭스 그라데이션

      const outerGlowFilter = ctx.createRadialGradient(xCenter, yCenter, targetPixelRadius * 0.25, xCenter, yCenter, dynamicCoronaRadius);

      outerGlowFilter.addColorStop(0, `rgba(${rgbString}, 0.58)`);

      outerGlowFilter.addColorStop(0.25, `rgba(${rgbString}, 0.33)`);

      outerGlowFilter.addColorStop(0.65, `rgba(${rgbString}, 0.06)`);

      outerGlowFilter.addColorStop(1, 'rgba(0, 0, 0, 0)');


      ctx.beginPath();

      ctx.arc(xCenter, yCenter, dynamicCoronaRadius, 0, Math.PI * 2);

      ctx.fillStyle = outerGlowFilter;

      ctx.fill();


      // 내부 인텐스 가스 광구 필터

      const innerGlowFilter = ctx.createRadialGradient(xCenter, yCenter, targetPixelRadius * 0.55, xCenter, yCenter, targetPixelRadius + 16);

      innerGlowFilter.addColorStop(0, `rgba(${rgbString}, 1.0)`);

      innerGlowFilter.addColorStop(0.55, `rgba(${rgbString}, 0.45)`);

      innerGlowFilter.addColorStop(1, 'rgba(0, 0, 0, 0)');


      ctx.beginPath();

      ctx.arc(xCenter, yCenter, targetPixelRadius + 16, 0, Math.PI * 2);

      ctx.fillStyle = innerGlowFilter;

      ctx.fill();


      // 3. 주 항성 본체 광구(Photosphere) 코어 라인 구동

      ctx.beginPath();

      ctx.arc(xCenter, yCenter, targetPixelRadius, 0, Math.PI * 2);

      ctx.fillStyle = computedRGB.hex;

      ctx.fill();


      // 4. 이중 렌더링 이펙트 - 초고온 열역학 화이트 코어 포화 그라데이션 필터

      const centralWhiteCore = ctx.createRadialGradient(xCenter, yCenter, 0, xCenter, yCenter, targetPixelRadius * 0.78);

      centralWhiteCore.addColorStop(0, '#ffffff'); // 중심부 열역학적 포화 고광도 화이트 아웃 백색화

      centralWhiteCore.addColorStop(0.35, `rgba(${rgbString}, 0.96)`);

      centralWhiteCore.addColorStop(1, computedRGB.hex);


      ctx.beginPath();

      ctx.arc(xCenter, yCenter, targetPixelRadius, 0, Math.PI * 2);

      ctx.fillStyle = centralWhiteCore;

      ctx.fill();


      // 애니메이션 프레임 루프 순환 연동

      animationId = requestAnimationFrame(renderStellarPhotosphere);

    }
